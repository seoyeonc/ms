def fnPeriodSeparate(lastDate, nTrainLen, nValidLen, nTestLen):
    '''
    Argument
    --------- + ---------------------------
    lastDate  | 주어진 데이터의 마지막 시점
    nTrainLen | 훈련기간 일 수
    nValidLen | 훈련기간 일 수
    nTestLen  | 테스트기간 일 수
    --------- + ---------------------------    
    '''
    nTrainLen -= 1
    nValidLen -= 1
    nTestLen -= 1

    TEST_END    = lastDate    
    TEST_START  = pd.date_range(TEST_END, periods = 2, freq = '-{}B'.format(nTestLen)).strftime('%Y-%m-%d')[1]
    VALID_END   = pd.date_range(TEST_START, periods = 2, freq = '-1B').strftime('%Y-%m-%d')[1]
    VALID_START = pd.date_range(VALID_END, periods = 2, freq = '-{}B'.format(nValidLen)).strftime('%Y-%m-%d')[1]
    TRAIN_END   = pd.date_range(VALID_START, periods = 2, freq = '-1B').strftime('%Y-%m-%d')[1]
    TRAIN_START = pd.date_range(TRAIN_END, periods = 2, freq = '-{}B'.format(nTrainLen)).strftime('%Y-%m-%d')[1]
    print('training  기간 :', TRAIN_START,'~', TRAIN_END)
    print('valid 기간     :', VALID_START, '~', VALID_END )
    print('test 기간      :', TEST_START,'~', TEST_END )
    return TRAIN_START, TRAIN_END, VALID_START, VALID_END, TEST_START, TEST_END

def fnReshapeDF(normDF, yVarName, nStep, nInterval):    
    '''
     Argument
    ------------ + ---------------------------
    normDF       |  적용 데이터
    yVarName     |  목표변수명
    nStep        |  시퀀스의 갯 수
    nInterval    |  시퀀스 간의 간격
    ------------ + ---------------------------    
    '''    
    xL = []
    yL = []
    idxDateDF = pd.DataFrame()
    startIdx = 0 + nInterval*(nStep - 1)
    endIdx = normDF.shape[0]
    
    normDF = normDF.copy()
    yNP = np.array(normDF.pop(yVarName))
    xNP = np.array(normDF)
    
    
    for i in range(startIdx , endIdx):
        indices = range(i, i - (nInterval*nStep), -nInterval)
        sortIdx = sorted(list(indices))

        # x
        xL.append(xNP[indices])

        # y
        yL.append(yNP[i])        

        # dateDF
        tmpDateDF = dateDF.iloc[i:(i+1), :]
        idxDateDF = pd.concat([idxDateDF, tmpDateDF], axis = 0)
    
    idxDateDF = idxDateDF.reset_index(drop = True) 
    
    xTotNP = np.array(xL) 
    yTotNP = np.array(yL).reshape((-1, 1))
    return xTotNP, yTotNP, idxDateDF
